# server/cli/esp32/main/CMakeLists.txt — the ESP-IDF "main" component.
#
# Compiles app_main.cpp (wifi/DHCP bring-up) + the CatchChallenger all-in-one
# game server, in the no-filesystem "datapack-cpp" profile: datapack + settings
# in flash (.rodata), in-memory DB, select(2) over lwIP, no Qt, no XML.
#
# ── INTEGRATION POINT ────────────────────────────────────────────────────────
# The server source set is assembled below from the SAME files the host CLI
# server builds (server/cli/CMakeLists.txt `_filedb_sources` + the
# catchchallenger_general / catchchallenger_server_base INTERFACE libs from
# general/general.cmake + server/catchchallenger-server.cmake), MINUS everything
# the ESP32 profile can't use (XML datapack loaders, SQL DB drivers, the other
# server mains, Qt, the clients, tools). If a refactor moves sources, adjust the
# globs/excludes here; test/testingcompilationESP32.py (host stage1/stage2) is
# the source of truth for which TUs must compile in this profile. See ESP32.md.

get_filename_component(CC_REPO "${CMAKE_CURRENT_LIST_DIR}/../../../.." ABSOLUTE)

# Human-readable datapack + settings, generated by stage1 (server-cli --save).
# Passed on the idf.py command line: -DCATCHCHALLENGER_DATAPACK_CPP_DIR=<dir>.
if(NOT CATCHCHALLENGER_DATAPACK_CPP_DIR)
    set(CATCHCHALLENGER_DATAPACK_CPP_DIR "/mnt/data/perso/tmpfs/datapack-cpp")
endif()

# --- shared + server game-logic sources (globbed, then filtered) -------------
file(GLOB_RECURSE _cc_general    "${CC_REPO}/general/base/*.cpp"
                                 "${CC_REPO}/general/fight/*.cpp")
file(GLOB         _cc_blake3     "${CC_REPO}/general/blake3/*.c")
# server/base AND its siblings server/fight + server/crafting: the latter two
# define the Client fight/crafting method overrides (Client::tryEscape,
# tryCapture, takeAnObjectOnMap, …) that server/base/ calls UNCONDITIONALLY, so
# omitting them is an undefined-reference link error. (catchchallenger-server.cmake
# lists all three in catchchallenger_server_base.)
file(GLOB_RECURSE _cc_serverbase "${CC_REPO}/server/base/*.cpp"
                                 "${CC_REPO}/server/fight/*.cpp"
                                 "${CC_REPO}/server/crafting/*.cpp")
# Drop ONLY what this profile genuinely excludes (mirrors general.cmake's NOXML
# branch + catchchallenger-server.cmake): vendored tinyXML2, Qt TUs, and the SQL
# backend (PreparedDBQuery*/SqlFunction/EventLoop{Postgresql,MySQL,Database} —
# DB_INTERNAL_VARS reuses the FileDB record path, no SQL). The XML *loaders*
# (Map_loader*/DatapackGeneralLoader*/FightLoader*) ARE compiled under NOXML
# (they carry #ifdef CATCHCHALLENGER_NOXML stubs) — do NOT exclude them.
list(FILTER _cc_general    EXCLUDE REGEX "/tinyXML2/|/Qt|Qt\\.cpp$")
list(FILTER _cc_serverbase EXCLUDE REGEX "PreparedDBQuery|SqlFunction|DatabaseProtocol|EventLoop(Postgresql|MySQL|Database)|/Qt|Qt\\.cpp$")

# --- the all-in-one CLI event-loop server (explicit, mirrors _filedb_sources) -
set(_cc_cli
    ${CC_REPO}/server/cli/SocketUtil.cpp
    ${CC_REPO}/server/cli/EventLoopClient.cpp
    ${CC_REPO}/server/cli/EventLoopClientList.cpp
    ${CC_REPO}/server/cli/EventLoopServer.cpp
    ${CC_REPO}/server/cli/EventLoop.cpp
    ${CC_REPO}/server/cli/EventLoopTimer.cpp
    ${CC_REPO}/server/cli/EventLoopGenericServer.cpp
    ${CC_REPO}/server/cli/BaseServerEventLoop.cpp
    ${CC_REPO}/server/cli/ServerPrivateVariablesEventLoop.cpp
    ${CC_REPO}/server/cli/ClientWithMapEventLoop.cpp
    ${CC_REPO}/server/cli/main-unix.cpp
    ${CC_REPO}/server/cli/main-unix2.cpp
    ${CC_REPO}/server/cli/timer/TimerCityCapture.cpp
    ${CC_REPO}/server/cli/timer/TimerSendInsertMoveRemove.cpp
    ${CC_REPO}/server/cli/timer/TimerPositionSync.cpp
    ${CC_REPO}/server/cli/timer/TimerBroadcastAnnounce.cpp
    ${CC_REPO}/server/cli/timer/TimerDdos.cpp
    ${CC_REPO}/server/cli/timer/TimerEvents.cpp
    ${CC_REPO}/server/cli/timer/TimeRangeEvent.cpp
    ${CC_REPO}/server/cli/timer/PlayerUpdaterEventLoop.cpp
    ${CC_REPO}/server/cli/timer/TimeRangeEventScan.cpp
)

set(_cc_datapack_cpp "${CATCHCHALLENGER_DATAPACK_CPP_DIR}/datapack_cpp.cpp")

idf_component_register(
    SRCS "app_main.cpp"
         ${_cc_cli} ${_cc_general} ${_cc_blake3} ${_cc_serverbase}
         "${_cc_datapack_cpp}"
    INCLUDE_DIRS "." "${CC_REPO}" "${CC_REPO}/server/cli" "${CC_REPO}/server/base"
                 "${CATCHCHALLENGER_DATAPACK_CPP_DIR}"
    REQUIRES esp_wifi esp_netif esp_event nvs_flash lwip
)

# The CLI's int main() becomes catchchallenger_cli_main() (app_main calls it).
set_source_files_properties(${CC_REPO}/server/cli/main-unix.cpp
    PROPERTIES COMPILE_DEFINITIONS "main=catchchallenger_cli_main"
    DIRECTORY "${CMAKE_CURRENT_LIST_DIR}")

# no-filesystem datapack-cpp profile defines (mirror the host stage2 build).
target_compile_definitions(${COMPONENT_LIB} PRIVATE
    CATCHCHALLENGER_SERVER
    QT_NO_EMIT
    CATCHCHALLENGER_CLASS_ALLINONESERVER
    CATCHCHALLENGER_CACHE_HPS
    CATCHCHALLENGER_DB_FILE
    CATCHCHALLENGER_DB_INTERNAL_VARS
    CATCHCHALLENGER_NOXML
    CATCHCHALLENGER_SELECT
    CATCHCHALLENGER_DATAPACK_CPP
    CC_TARGET_ESP32
)
# vendored blake3/xxhash/zstd/tinyxml2: the cross sysroot has no system copy.
target_compile_options(${COMPONENT_LIB} PRIVATE -Os -fno-exceptions -fno-rtti)
